actions = ['u','r','d','l']

def choose_action(epsilon):

    action = None
    
    # 
    if random.random() >= epsilon:
        # 
        action = actions[random.randint(0,len(actions)-1)]
    #  Q 
    else:
        action = max(qline, key=qline.get)
    return action
